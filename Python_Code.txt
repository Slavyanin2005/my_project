def greet_user():
    """Функция приветствия пользователя."""
    print("Привет! Эта программа умеет здороваться.")
    user_name = input("Как тебя зовут? ").strip()
    
    greeting = f"Привет, {user_name}!" if user_name else "Привет, аноним!"
    print(f"{greeting} Добро пожаловать!")


def main():
    """Основная функция программы."""
    greet_user()
    print("Программа завершена. Можешь добавить сюда новую логику!")


if __name__ == "__main__":
    main()
